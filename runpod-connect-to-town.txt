#!/usr/bin/env python3
import json, torch, psycopg2, time, psutil
from datetime import datetime

class RunPodReporter:
    def __init__(self):
        self.db_config = {
            "host": "aurora-postgres-u44170.vm.elestio.app",
            "port": 25432,
            "dbname": "aurora_unified",
            "user": "aurora_app", 
            "password": "TestPilot2025_Aurora!",
            "sslmode": "require"
        }
        self.gpu_host = "aurora-gpu-worker"
    
    def get_db_connection(self):
        return psycopg2.connect(**self.db_config)
    
    def get_gpu_status(self):
        status = {
            "timestamp": datetime.utcnow().isoformat(),
            "status": "healthy",
            "gpu": {},
            "system": {}
        }
        
        # GPU Check
        try:
            status["gpu"] = {
                "available": torch.cuda.is_available(),
                "count": torch.cuda.device_count(),
                "name": torch.cuda.get_device_name(0) if torch.cuda.is_available() else None,
                "memory_total": torch.cuda.get_device_properties(0).total_memory if torch.cuda.is_available() else 0,
                "memory_allocated": torch.cuda.memory_allocated(0) if torch.cuda.is_available() else 0
            }
        except Exception as e:
            status["gpu"] = {"error": str(e)}
            status["status"] = "degraded"
        
        # System Check
        try:
            status["system"] = {
                "cpu_percent": psutil.cpu_percent(),
                "memory_percent": psutil.virtual_memory().percent,
                "disk_percent": psutil.disk_usage('/').percent
            }
        except Exception as e:
            status["system"] = {"error": str(e)}
        
        return status
    
    def report_status(self):
        try:
            gpu_status = self.get_gpu_status()
            
            conn = self.get_db_connection()
            with conn.cursor() as cur:
                cur.execute("""
                    INSERT INTO gpu_status (gpu_host, status, owner_id)
                    VALUES (%s, %s, %s)
                """, (self.gpu_host, json.dumps(gpu_status), "aurora"))
            
            conn.commit()
            conn.close()
            
            print(f"âœ… Status reported: {gpu_status['status']}")
            return True
            
        except Exception as e:
            print(f"âŒ Error reporting status: {e}")
            return False
    
    def run_continuous_reporting(self, interval=30):
        print(f"ðŸš€ Starting continuous GPU status reporting (every {interval}s)")
        print("Press Ctrl+C to stop")
        
        try:
            while True:
                success = self.report_status()
                if not success:
                    print("âš ï¸ Status report failed, retrying...")
                
                time.sleep(interval)
                
        except KeyboardInterrupt:
            print("\nðŸ›‘ Reporting stopped by user")

if __name__ == "__main__":
    reporter = RunPodReporter()
    
    print("ðŸš€ RUNPOD TOWN REPORTER")
    print("=" * 25)
    
    # Test single report
    print("ðŸ” Testing single status report...")
    success = reporter.report_status()
    
    if success:
        print("âœ… Single report successful!")
        print("\nðŸ”„ Starting continuous reporting...")
        reporter.run_continuous_reporting()
    else:
        print("âŒ Single report failed - check configuration")
EOF
